72      @Test
73      public void testWriteFiles() throws IOException {
74          LOG.info("Testing writeFiles method...");
75  
76          String targetDir = "target/test/filetool";
77          FileTool fileTool = new FileTool(targetDir, new String[] {"htm", "html"}, true);
78          Map<String, String> map = new HashMap<String, String>();
79          map.put("file.html", "root file");
80          map.put("/file2.html", "another root file");
81          map.put("file3.html/", "another root file like folder");
82          map.put("/template/file.html", "template file");
83          map.put("template/file01.html", "template file 01");
84          map.put("/template/subfolder/file.html", "template subfolder file");
85          fileTool.writeFiles(map, targetDir);
86  
87          Map<String, String> files = fileTool.getFiles();
88          Assertions.assertTrue(files.containsKey("file.html"));
89          Assertions.assertTrue(files.containsKey("file2.html"));
90          Assertions.assertTrue(files.containsKey("file3.html"));
91          Assertions.assertTrue(files.containsKey("template/file.html"));
92          Assertions.assertTrue(files.containsKey("template/file01.html"));
93          Assertions.assertTrue(files.containsKey("template/subfolder/file.html"));
94  
95          // TODO: check file exists and its contents is written correctly
96  
97          LOG.info("Passed");
98      }
