298     @Override
299     public void execute() throws MojoExecutionException {
300         if (!enabled) {
301             getLog().info("HTML compression was turned off.");
302             return;
303         }
304 
305         getLog().info("Compressing " + srcFolder);
306         HtmlCompressor htmlCompressor = new HtmlCompressor(srcFolder, targetFolder);
307 
308         htmlCompressor.setFileExt(fileExt);
309         htmlCompressor.setFileEncoding(encoding);
310         htmlCompressor.setCreateJsonFile(javascriptHtmlSprite);
311         htmlCompressor.setJsonIntegrationFilePath(javascriptHtmlSpriteIntegrationFile);
312         htmlCompressor.setTargetJsonFilePath(javascriptHtmlSpriteTargetFile);
313 
314         com.googlecode.htmlcompressor.compressor.HtmlCompressor htmlCompressorHandler = new com.googlecode.htmlcompressor.compressor.HtmlCompressor();
315         htmlCompressorHandler.setEnabled(enabled);
316         htmlCompressorHandler.setRemoveComments(removeComments);
317         htmlCompressorHandler.setRemoveMultiSpaces(removeMultiSpaces);
318         htmlCompressorHandler.setRemoveIntertagSpaces(removeIntertagSpaces);
319         htmlCompressorHandler.setRemoveQuotes(removeQuotes);
320         htmlCompressorHandler.setSimpleDoctype(simpleDoctype);
321         htmlCompressorHandler.setRemoveScriptAttributes(removeScriptAttributes);
322         htmlCompressorHandler.setRemoveStyleAttributes(removeStyleAttributes);
323         htmlCompressorHandler.setRemoveLinkAttributes(removeLinkAttributes);
324         htmlCompressorHandler.setRemoveFormAttributes(removeFormAttributes);
325         htmlCompressorHandler.setRemoveInputAttributes(removeInputAttributes);
326         htmlCompressorHandler.setSimpleBooleanAttributes(simpleBooleanAttributes);
327         htmlCompressorHandler.setRemoveJavaScriptProtocol(removeJavaScriptProtocol);
328         htmlCompressorHandler.setRemoveHttpProtocol(removeHttpProtocol);
329         htmlCompressorHandler.setRemoveHttpsProtocol(removeHttpsProtocol);
330         htmlCompressorHandler.setCompressCss(compressCss);
331         htmlCompressorHandler.setPreserveLineBreaks(preserveLineBreaks);
332         htmlCompressorHandler.setYuiCssLineBreak(yuiCssLineBreak);
333         htmlCompressorHandler.setCompressJavaScript(compressJavaScript);
334         htmlCompressorHandler.setYuiJsNoMunge(yuiJsNoMunge);
335         htmlCompressorHandler.setYuiJsPreserveAllSemiColons(yuiJsPreserveAllSemiColons);
336         htmlCompressorHandler.setYuiJsLineBreak(yuiJsLineBreak);
337         htmlCompressorHandler.setYuiJsDisableOptimizations(yuiJsDisableOptimizations);
338         htmlCompressorHandler.setGenerateStatistics(generateStatistics);
339 
340         if (jsCompressor.equalsIgnoreCase("closure")) {
341             ClosureJavaScriptCompressor closureCompressor = new ClosureJavaScriptCompressor();
342             if (closureOptLevel != null && closureOptLevel.equalsIgnoreCase(ClosureJavaScriptCompressor.COMPILATION_LEVEL_ADVANCED)) {
343                 closureCompressor.setCompilationLevel(CompilationLevel.ADVANCED_OPTIMIZATIONS);
344                 closureCompressor.setCustomExternsOnly(closureCustomExternsOnly != null);
345                 if(closureExterns.length  > 0) {
346                     List<SourceFile> externs = new ArrayList<SourceFile>();
347                     for(String externFile : closureExterns) {
348                         externs.add(SourceFile.fromFile(externFile));
349                     }
350                     closureCompressor.setExterns(externs);
351                 }
352             } else if (closureOptLevel != null && closureOptLevel.equalsIgnoreCase(ClosureJavaScriptCompressor.COMPILATION_LEVEL_WHITESPACE)) {
353                 closureCompressor.setCompilationLevel(CompilationLevel.WHITESPACE_ONLY);
354             } else {
355                 closureCompressor.setCompilationLevel(CompilationLevel.SIMPLE_OPTIMIZATIONS);
356             }
357 
358             htmlCompressorHandler.setJavaScriptCompressor(closureCompressor);
359         }
360 
361         List<Pattern> preservePatternList = new ArrayList<Pattern>();
362         boolean phpTagPatternAdded = false;
363         boolean serverScriptTagPatternAdded = false;
364         if (predefinedPreservePatterns != null) {
365             for (String pattern : predefinedPreservePatterns) {
366                 if (!phpTagPatternAdded && pattern.equalsIgnoreCase("PHP_TAG_PATTERN")) {
367                     preservePatternList.add(com.googlecode.htmlcompressor.compressor.HtmlCompressor.PHP_TAG_PATTERN);
368                     phpTagPatternAdded = true;
369                 } else if (!serverScriptTagPatternAdded && pattern.equalsIgnoreCase("SERVER_SCRIPT_TAG_PATTERN")) {
370                     preservePatternList.add(com.googlecode.htmlcompressor.compressor.HtmlCompressor.SERVER_SCRIPT_TAG_PATTERN);
371                     serverScriptTagPatternAdded = true;
372                 }
373             }
374         }
375         if (preservePatterns != null) {
376             for (String preservePatternString : preservePatterns) {
377                 if (!preservePatternString.isEmpty()) {
378                     try {
379                         preservePatternList.add(Pattern.compile(preservePatternString));
380                     } catch (PatternSyntaxException e) {
381                         throw new MojoExecutionException(e.getMessage());
382                     }
383                 }
384             }
385         }
386         if (preservePatternFiles != null) {
387             for (File file : preservePatternFiles) {
388                 try {
389                     List<String> fileLines = Files.readAllLines(file.toPath(), Charset.forName(encoding));
390                     for (String line : fileLines) {
391                         if (!line.isEmpty()) {
392                             preservePatternList.add(Pattern.compile(line));
393                         }
394                     }
395                 } catch (PatternSyntaxException e) {
396                     throw new MojoExecutionException(e.getMessage());
397                 } catch (IOException e) {
398                     throw new MojoExecutionException(e.getMessage());
399                 }
400             }
401         }
402         htmlCompressorHandler.setPreservePatterns(preservePatternList);
403         htmlCompressor.setHtmlCompressor(htmlCompressorHandler);
404 
405         try {
406             htmlCompressor.compress();
407         } catch(Exception e) {
408             throw new MojoExecutionException(e.getMessage());
409         }
410 
411         boolean si = true;
412 
413         // TODO: if no files matched pattern (*.htm or *.html) then this gives NullPointerException
414         int origFilesizeBytes = htmlCompressor.getHtmlCompressor().getStatistics().getOriginalMetrics().getFilesize();
415         String origFilesize = FileTool.humanReadableByteCount(origFilesizeBytes, si);
416         String origEmptyChars = String.valueOf(htmlCompressor.getHtmlCompressor().getStatistics().getOriginalMetrics().getEmptyChars());
417         String origInlineEventSize = FileTool.humanReadableByteCount(htmlCompressor.getHtmlCompressor().getStatistics().getOriginalMetrics().getInlineEventSize(), si);
418         String origInlineScriptSize = FileTool.humanReadableByteCount(htmlCompressor.getHtmlCompressor().getStatistics().getOriginalMetrics().getInlineScriptSize(), si);
419         String origInlineStyleSize = FileTool.humanReadableByteCount(htmlCompressor.getHtmlCompressor().getStatistics().getOriginalMetrics().getInlineStyleSize(), si);
420 
421         int compFilesizeBytes = htmlCompressor.getHtmlCompressor().getStatistics().getCompressedMetrics().getFilesize();
422         String compFilesize = FileTool.humanReadableByteCount(compFilesizeBytes, si);
423         String compEmptyChars = String.valueOf(htmlCompressor.getHtmlCompressor().getStatistics().getCompressedMetrics().getEmptyChars());
424         String compInlineEventSize = FileTool.humanReadableByteCount(htmlCompressor.getHtmlCompressor().getStatistics().getCompressedMetrics().getInlineEventSize(), si);
425         String compInlineScriptSize = FileTool.humanReadableByteCount(htmlCompressor.getHtmlCompressor().getStatistics().getCompressedMetrics().getInlineScriptSize(), si);
426         String compInlineStyleSize = FileTool.humanReadableByteCount(htmlCompressor.getHtmlCompressor().getStatistics().getCompressedMetrics().getInlineStyleSize(), si);
427 
428         String elapsedTime = FileTool.getElapsedHMSTime(htmlCompressor.getHtmlCompressor().getStatistics().getTime());
429         String preservedSize = FileTool.humanReadableByteCount(htmlCompressor.getHtmlCompressor().getStatistics().getPreservedSize(), si);
430         Float compressionRatio = Float.valueOf(compFilesizeBytes) / Float.valueOf(origFilesizeBytes);
431         Float spaceSavings = Float.valueOf(1) - compressionRatio;
432 
433         String format = "%-30s%-30s%-30s%-2s";
434         NumberFormat formatter = new DecimalFormat("#0.00");
435         String eol = "\n";
436         String hr = "+-----------------------------+-----------------------------+-----------------------------+";
437         StringBuilder sb = new StringBuilder("HTML compression statistics:").append(eol);
438         sb.append(hr).append(eol);
439         sb.append(String.format(format, "| Category", "| Original", "| Compressed", "|")).append(eol);
440         sb.append(hr).append(eol);
441         sb.append(String.format(format, "| Filesize", "| " + origFilesize, "| " + compFilesize, "|")).append(eol);
442         sb.append(String.format(format, "| Empty Chars", "| " + origEmptyChars, "| " + compEmptyChars, "|")).append(eol);
443         sb.append(String.format(format, "| Script Size", "| " + origInlineScriptSize, "| " + compInlineScriptSize, "|")).append(eol);
444         sb.append(String.format(format, "| Style Size", "| " + origInlineStyleSize, "| " + compInlineStyleSize, "|")).append(eol);
445         sb.append(String.format(format, "| Event Handler Size", "| " + origInlineEventSize, "| " + compInlineEventSize, "|")).append(eol);
446         sb.append(hr).append(eol);
447         sb.append(String.format("%-90s%-2s",
448                 String.format("| Time: %s, Preserved: %s, Compression Ratio: %s, Savings: %s%%",
449                         elapsedTime, preservedSize, formatter.format(compressionRatio), formatter.format(spaceSavings*100)),
450                 "|")).append(eol);
451         sb.append(hr).append(eol);
452 
453         String statistics = sb.toString();
454         getLog().info(statistics);
455         try {
456             FileUtils.writeStringToFile(new File(htmlCompressionStatistics), statistics, encoding);
457         } catch (IOException e) {
458             throw new MojoExecutionException(e.getMessage());
459         }
460 
461         getLog().info("HTML compression completed.");
462     }
